BASIC USAGE (Copy & Paste Ready)
================================

import { createSSIProvider } from './market-data/ssi.provider.js';
import { createMarketDataClient } from './market-data/client.js';

// Create client
const client = createMarketDataClient([createSSIProvider()]);

// Get single stock
const stock = await client.getStockPrice('ACB');
console.log(`${stock.symbol}: ${stock.currentPrice}`);

// Get price table
const table = await client.getPriceTable('VN30');
table.stocks.forEach(s => console.log(`${s.symbol}: ${s.currentPrice}`));

// Get multiple indices
const indices = await client.getMultipleIndices(['VNINDEX','VN30','HNX30']);
indices.forEach(i => console.log(`${i.indexCode}: ${i.currentValue}`));

// Use cache
console.log(client.getCacheStats());  // View cache performance

==================================

getStockPrice(symbol)              → Get single stock
